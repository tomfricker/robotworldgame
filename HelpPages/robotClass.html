public class Robot

{

    //the age of the robot

    private int age;

    //the name of the robot

    private String name;

    //the hat of the robot

    private Item hat;

    //if the robot has a buddy

    private boolean hasBuddy;

    //if the robot is hungry

    private boolean isHungry;



    /**

     * Constructor for objects of class Robot with parameters

     */

    public Robot(String fullName, int robotAge)

    {

        // initialise the fields

        name = fullName;

        age = robotAge;

        hat = new Item();

        hasBuddy = true;

        isHungry = true;

    }



    /**

     * Constructor for objects of class Robot without parameters

     */

    public Robot()

    {

        // initialise the fields

        name = "Robbie";

        age = 15;

        hat = new Item();

        hasBuddy = true;

        isHungry = true;

    }



    /**

     * Returns the age of the robot

     *

     * @return    the age of the robot

     */

    public int getAge()

    {

        return age;

    }

    

    /**

     * Sets the age of the robot to the parameter value

     *

     * @param  newAge  the age you're setting for the robot

     */

    public void setAge(int newAge)

    {

        age = newAge;

    }

    

    /**

     * Returns the name of the robot

     *

     * @return    the name of the robot

     */

    public String getName()

    {

        return name;

    }

    

    /**

     * Changes the name of the robot to the parameter value
 
     *

     * @param  newName  the new name of the robot

     */

    public void changeName(String newName)

    {

        name = newName;

    }

    

    /**

     * Returns whether the robot has a buddy or not

     *

     * @return   the hasBuddy value of the robot

     */

    public boolean getBuddy()

    {

        return hasBuddy;

    }

    

    /**

     * Returns whether the robot is hungry or not

     *

     * @return   the isHungry value of the robot

     */

    public boolean getHungry()

    {

        return isHungry;

    }

    

    /**

     * Set the hat colour of the robot

     *

     * @param  colour  the colour you're setting for the hat

     */

    public void setHatColour(String colour)

    {

        hat.changeColour("blue");

    }

    

    /**

     * The robot eats

     *

     * @param  fruit  the Item that the robot eats

     */

    public void eat(Item fruit)

    {

        fruit.disappear();

        isHungry = false;

    }

    

}
